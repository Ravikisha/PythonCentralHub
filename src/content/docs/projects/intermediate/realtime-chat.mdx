### üé® Example Usage

```python title="realtimechat.py" showLineNumbers{1}
# Initialize chat application
app = RealtimeChatApp()

# Create a new user
db = ChatDatabase()
user_id = db.create_user(
    username="johndoe",
    email="john@example.com", 
    password="secure_password",
    display_name="John Doe"
)

# Authenticate user
user = db.authenticate_user("johndoe", "secure_password")
print(f"User authenticated: {user['display_name']}")

# Get user's chat rooms
rooms = db.get_user_rooms(user_id)
for room in rooms:
    print(f"Room: {room['name']} ({room['type']})")

# Get room messages
messages = db.get_room_messages(room_id=1, limit=20)
for message in messages:
    print(f"{message['display_name']}: {message['content']}")

# Run the application
app.run(host='localhost', port=5000)
```
