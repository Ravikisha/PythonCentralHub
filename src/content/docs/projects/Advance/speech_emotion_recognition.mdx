```python title="speech_emotion_recognition.py" showLineNumbers{1}
def extract_features(audio_path):
    y, sr = librosa.load(audio_path)
    mfccs = np.mean(librosa.feature.mfcc(y=y, sr=sr).T, axis=0)
    return mfccs

def train_model(X, y):
    model = RandomForestClassifier()
    model.fit(X, y)
    return model
```
