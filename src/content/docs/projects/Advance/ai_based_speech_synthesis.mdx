
2. **Text-to-Speech Function**
```python title="ai_based_speech_synthesis.py" showLineNumbers{1}
def speak_text(text):
    engine = pyttsx3.init()
    engine.say(text)
    engine.runAndWait()

def save_speech(text, lang='en', filename='output.mp3'):
    tts = gTTS(text=text, lang=lang)
    tts.save(filename)
```

3. **CLI Interface and Error Handling**
```python title="ai_based_speech_synthesis.py" showLineNumbers{1}
def main():
    print("AI-based Speech Synthesis")
    while True:
        cmd = input('> ')
        if cmd == 'speak':
            text = input("Text: ")
            try:
                speak_text(text)
            except Exception as e:
                print(f"Error: {e}")
        elif cmd == 'save':
            text = input("Text: ")
            lang = input("Language (default 'en'): ") or 'en'
            filename = input("Filename (default 'output.mp3'): ") or 'output.mp3'
            try:
                save_speech(text, lang, filename)
                print(f"Saved to {filename}")
            except Exception as e:
                print(f"Error: {e}")
        elif cmd == 'exit':
            break
        else:
            print("Unknown command. Type 'speak', 'save', or 'exit'.")

if __name__ == "__main__":
    main()
```
