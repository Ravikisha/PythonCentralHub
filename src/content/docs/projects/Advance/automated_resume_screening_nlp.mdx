```python title="automated_resume_screening_nlp.py" showLineNumbers{1}
def rank_candidates(resumes, job_desc):
    vectorizer = TfidfVectorizer()
    X = vectorizer.fit_transform(resumes + [job_desc])
    scores = (X[:-1] * X[-1].T).toarray().flatten()
    ranked = sorted(zip(resumes, scores), key=lambda x: x[1], reverse=True)
    return ranked
```
