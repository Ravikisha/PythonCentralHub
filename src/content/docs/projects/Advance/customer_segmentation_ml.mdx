```python title="customer_segmentation_ml.py" showLineNumbers{1}
def segment_customers(df):
    kmeans = KMeans(n_clusters=3)
    df['Segment'] = kmeans.fit_predict(df)
    return df

def plot_segments(df):
    plt.scatter(df.iloc[:,0], df.iloc[:,1], c=df['Segment'])
    plt.show()
```
