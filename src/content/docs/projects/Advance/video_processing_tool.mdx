```python title="video_processing_tool.py" showLineNumbers{1}
def extract_frames(video_path):
    cap = cv2.VideoCapture(video_path)
    count = 0
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break
        cv2.imwrite(f'frame_{count}.jpg', frame)
        count += 1
    cap.release()
```
