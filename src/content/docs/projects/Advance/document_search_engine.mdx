```python title="document_search_engine.py" showLineNumbers{1}
def index_documents(docs):
    vectorizer = TfidfVectorizer()
    X = vectorizer.fit_transform(docs)
    return vectorizer, X

def search_query(query, vectorizer, X):
    q_vec = vectorizer.transform([query])
    scores = (X * q_vec.T).toarray().flatten()
    return scores
```
