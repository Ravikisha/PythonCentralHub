### Code Breakdown
1. **Import Libraries and Define Intents**
```python title="advanced_chatbot_with_nlp.py" showLineNumbers{1}
import nltk
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
nltk.download('punkt')

intents = {
    'greet': ['hello', 'hi', 'hey'],
    'bye': ['bye', 'goodbye', 'see you'],
    'order': ['order pizza', 'I want pizza', 'pizza please'],
    'thanks': ['thanks', 'thank you'],
}
```

2. **Intent Classifier Setup**
```python title="advanced_chatbot_with_nlp.py" showLineNumbers{1}
vectorizer = CountVectorizer()
X = []
y = []
for intent, examples in intents.items():
    X.extend(examples)
    y.extend([intent] * len(examples))
X_vec = vectorizer.fit_transform(X)
clf = MultinomialNB()
clf.fit(X_vec, y)
```

3. **Context Management and Response Logic**
```python title="advanced_chatbot_with_nlp.py" showLineNumbers{1}
class Chatbot:
    def __init__(self):
        self.context = {}
    def get_intent(self, text):
        vec = vectorizer.transform([text])
        return clf.predict(vec)[0]
    def respond(self, text):
        intent = self.get_intent(text)
        if intent == 'greet':
            return "Hello! How can I help you?"
        elif intent == 'order':
            self.context['order'] = True
            return "What type of pizza would you like?"
        elif intent == 'thanks':
            return "You're welcome!"
        elif intent == 'bye':
            return "Goodbye!"
        else:
            return "Sorry, I didn't understand."
```

4. **CLI Interface and Error Handling**
```python title="advanced_chatbot_with_nlp.py" showLineNumbers{1}
def main():
    bot = Chatbot()
    print("Advanced Chatbot with NLP")
    while True:
        try:
            user_input = input('You: ')
            if user_input.lower() in ['exit', 'quit']:
                print("Bot: Goodbye!")
                break
            response = bot.respond(user_input)
            print(f"Bot: {response}")
        except Exception as e:
            print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
