```python title="automated_news_aggregator.py" showLineNumbers{1}
import requests
from bs4 import BeautifulSoup
from sumy.parsers.plaintext import PlaintextParser
from sumy.nlp.tokenizers import Tokenizer
from sumy.summarizers.lsa import LsaSummarizer
import nltk
```

2. **Web Scraping and Summarization Functions**
```python title="automated_news_aggregator.py" showLineNumbers{1}
def scrape_news(url):
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    paragraphs = soup.find_all('p')
    text = ' '.join([p.get_text() for p in paragraphs])
    return text

def summarize_text(text, sentences_count=5):
    parser = PlaintextParser.from_string(text, Tokenizer('english'))
    summarizer = LsaSummarizer()
    summary = summarizer(parser.document, sentences_count)
    return ' '.join(str(sentence) for sentence in summary)
```

3. **CLI Interface and Error Handling**
```python title="automated_news_aggregator.py" showLineNumbers{1}
def main():
    print("Automated News Aggregator")
    while True:
        cmd = input('> ')
        if cmd == 'aggregate':
            url = input("News URL: ")
            try:
                text = scrape_news(url)
                summary = summarize_text(text)
                print(f"Summary: {summary}")
            except Exception as e:
                print(f"Error: {e}")
        elif cmd == 'exit':
            break
        else:
            print("Unknown command. Type 'aggregate' or 'exit'.")

if __name__ == "__main__":
    main()
```
